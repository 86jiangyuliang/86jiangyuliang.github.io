知识点：notepad 快捷键

    1. 说明
        # notepad 快捷键中，数字只支持横排数字键，不支持小键盘的按键
        # 自定义快捷键尽量以 Ctrl + Alt + Shift 的方式
    2. 文件 File
        # 新建
            (01) Ctrl + N                       <== 新建
        # 打开
            (02) Ctrl + O                       <== 打开
            (03) Ctrl + Alt + Shift + F10       <== 打开当前文件的目录（自定义）
        # 保存
            (08) Ctrl + S                       <== 保存
            (09) Ctrl + Alt + S                 <== 另存为
            (11) Ctrl + Shift + S               <== 全部保存
        # 关闭
            (13) Ctrl + W                       <== 关闭标签页
            (14) Ctrl + Shift + W               <== 关闭全部标签页
            (25) Ctrl + Shift + T               <== 恢复最近关闭的文件
    3. 编辑 Edit
        # 大小写
            (32) Ctrl + Shift + U               <== 全部大写
            (33) Ctrl + U                       <== 全部小写
            (34) Alt + U                        <== 单词首字母大写
            (35) Alt + Shift + U                <== 单词首字母大写
            (36) Ctrl + Alt + U                 <== 句首字母大写
            (37) Ctrl + Alt + Shift +  U        <== 句首字母大写
        # 行
            (42) Ctrl + J                       <== 合并行
            (43) Ctrl + Shift + Up              <== 上移当前行
            (44) Ctrl + Shift + Up              <== 下移当前行
            (46) Ctrl + Alt + Shift + Space     <== 删除全部空行，包括空白字符（自定义）
        # 排序
            (49)                                <== 升序排列文本
            (50)                                <== 降序排列文本
            (51)                                <== 降序排列整数
            (52)                                <== 降序排列整数
            (53)                                <== 降序排列浮点数
            (54)                                <== 降序排列浮点数
        # 注释
            (58) Ctrl + Q                       <== 添加/取消单行注释
            (59) Ctrl + K                       <== 添加单行注释
            (60) Ctrl + Shift + K               <== 取消单行注释
            (61) Ctrl + Shift + Q               <== 添加区块注释
            (62) Ctrl + Alt + Shift + Q         <== 取消区块注释（自定义）
        # 剪切板
            (89) Ctrl + Alt + Shift + Z         <== 剪切板历史（自定义）
    4. 搜索 Search
        # 搜索
            (92) Ctrl + F                       <== 打开搜索界面
            (94) F3                             <== 查找下一个匹配，搜索条件参照上一次搜索
            (95) Shift + F3                     <== 查找上一个匹配，搜索条件参照上一次搜索
            (96) Ctrl + F3                      <== 按照当前选定内容，查找下一个
            (97) Ctrl + Shift + F3              <== 按照当前选定内容，查找上一个
        # 替换
            (100) Ctrl + H                      <== 打开替换界面
        # 行定位
            (105) Ctrl + G                      <== 行定位
        # 括号定位
            (106) Ctrl + B                      <== 定位到匹配的括号，支持 () [] {} 三种，不支持 '' ""，不支持中文括号 （）
            (107) Ctrl + Alt + B                <== 选定当前括号中的全部文字，支持 () [] {} 三种，不支持 '' ""，不支持中文括号 （）
        # 书签
            (108) Ctrl + M                      <== 打开标记界面
            (132) Ctrl + F2                     <== 设置/取消书签
            (133) F2                            <== 转到下一个书签
            (134) Shift + F2                    <== 转到上一个书签
            (135) Ctrl + Alt + Shift + F2       <== 清除全部书签（自定义）
    5. 视图 view
        # 全局
            (144) F11                           <== 切换到全屏
            (150) Ctrl + Alt + Shift + 1        <== 显示/不显示空格和制表符（自定义）
        # 标签页
            (168) Ctrl + PageDown               <== 切换到下一个标签页
            (169) Ctrl + PageUp                 <== 切换到上一个标签页
            (170) Ctrl + Shift + PageDown       <== 当前标签页后移
            (171) Ctrl + Shift + PageUp         <== 当前标签页迁移
        # 层次
            (175) Alt + H                       <== 隐藏当前选定行
            (177) Alt + 0                       <== 折叠所有层次
            (178) Alt + Shift + 0               <== 展开所有层次
            (179) Ctrl + Alt + F                <== 折叠当前层次
            (180) Ctrl + Alt + Shift + F        <== 展开当前层次
    6. 运行命令
        # 浏览器
            (03) Ctrl + Alt + Shift + R         <== 在 chrome 中打开当前文件
            (06) Alt+ F3                        <== 打开 wiki 搜索
    7. Scintilla
        # 基本操作
            (01) Ctrl + X 或 Shift + Del        <== SCI_CUT，剪切
            (02) Ctrl + C 或 Ctrl + Ins         <== SCI_COPY，复制
            (03) Ctrl + V 或 Shift + Ins        <== SCI_PASTE，黏贴
            (04) Ctrl + A                       <== SCI_SELECTALL，全选
            (07) Ctrl + Z 或 Alt + Backspace    <== SCI_UNDO，撤销
            (08) Ctrl + Y 或 Ctrl + Shift + Z   <== SCI_REDO，还原
        # 制表符
            (10) Tab                            <== SCI_TAB，制表符
            (11) Shift + Tab                    <== SCI_BACKTAB，反向制表符
        # ZOOM IN/OUT
            (13) Ctrl + Num+                    <== SCI_ZOOMIN，放大
            (14) Ctrl + Num+                    <== SCI_ZOOMOUT，缩小
            (15) Ctrl + Num/                    <== SCI_SETZOOM，恢复默认大小
            (xx) Ctrl + 鼠标滚轮                <== 放大缩小
        # 窗口滚动
            (24) Ctrl + Down                    <== SCI_LINESCROLLDOWN，向下滚动一行
            (28) Ctrl + Up                      <== SCI_LINESCROLLUP，向上滚动一行
        # 定位光标到区块的首位
            (29) Ctrl + ]                       <== SCI_PARADOWN，向下定位到下一个区块的开头
            (30) Ctrl + Shift + ]               <== SCI_PARADOWNEXTEND，向下定位到下一个区块的开头，并选取中间的文字
            (31) Ctrl + [                       <== SCI_PARAUP，向上定位到本区块的开头，或上一个区块的开头
            (32) Ctrl + Shift + [               <== SCI_PARAUPEXTEND，向上定位到本区块的开头，或上一个区块的开头，并选取中间的文字
        # 列选取
            (35) Alt + Shift + Left             <== SCI_CHARLEFTRECTEXTEND，向左进行列选择
            (38) Alt + Shift + Right            <== SCI_CHARRIGHTRECTEXTEND，向右进行列选择
        # 按照单词横向移动光标
            (39) Ctrl + Lift                    <== SCI_WORDLEFT，向左移动一个单词的位置
            (40) Ctrl + Shift + Lift            <== SCI_WORDLEFTEXTEND，向左移动一个单词的位置，并选取中间的文字
            (41) Ctrl + Right                   <== SCI_WORDRIGHT，向右移动一个单词的位置
            (46) Ctrl + Shift + Right           <== SCI_WORDRIGHTEXTEND，向右移动一个单词的位置，并选取中间的文字
            (47) Ctrl + /                       <== SCI_WORDPARTLEFT，向左移动一个词根的位置
            (48) Ctrl + Shift + /               <== SCI_WORDPARTLEFTEXTEND，向左移动一个词根的位置，并选取中间的文字
            (49) Ctrl + \                       <== SCI_WORDPARTRIGHT，向右移动一个词根的位置
            (50) Ctrl + Shift + \               <== SCI_WORDPARTRIGHTEXTEND，向右移动一个词根的位置，并选取中间的文字
        # 移动光标到行首尾
            (61) Home                           <== SCI_VCHOMEWRAP，移动到行首
            (67) End                            <== SCI_LINEENDWRAP，移动号行尾
        # 移动光标到文件首尾
            (69) Ctrl + Home                    <== SCI_DOCUMENTSTART，移动到文件开头
            (71) Ctrl + End                     <== SCI_DOCUMENTEND，移动到文件结尾
        # 横向行内删除
            (85) Ctrl + Backspace               <== SCI_DELWORDLEFT，向左删除一个单词
            (86) Ctrl + Del                     <== SCI_DELWORDRIGHT，向右删除一个单词
            (87) Ctrl + Shift + Backspace       <== SCI_DELLINELEFT，向左删除到行首
            (88) Ctrl + Shift + Del             <== SCI_DELLINERIGHT，向右删除到行尾
        # 行编辑
            (16) Ctrl + D                       <== SCI_SELECTIONDUPLICATE，复制行
            (89) Ctrl + Shift + L               <== SCI_LINEDELETE，删除行
            (90) Ctrl + L                       <== SCI_LINECUT，剪切行
            (91) Ctrl + Shift + X               <== SCI_LINECOPY，复制行到剪切板
            (92) Ctrl + T                       <== SCI_LINETRANSPOSE，行换位置
